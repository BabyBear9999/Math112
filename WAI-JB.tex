\begin{prob} [Arithmetic] Simplify 
$$n=(1+1)^{1-1} \times (10)^{1+1}+(1-1)^{1+1} \times (10)^{1\times1}$$ 
and then evaluate $(-1)^n$

\end {prob}

\emph{Soluion. }
	%solution text
	$$n = (1+1)^0 \cdot (10)^2 + (0)^(2) \cdot (10)^{1}$$
	$$n = 1\cdot 100 + 0 \cdot 10$$
	$$n=100$$

	$$(-1)^n = (-1)^{100} = -1$$
